package com.library.repository;

import com.library.domain.Book;

public class BookRepository {
    private final List<Book> books = new ArrayList<>(); // —Å—Å—ã–ª–æ—á–Ω—ã–π —Ç–∏–ø

    // –ú–µ—Ç–æ–¥ —Å —Ü–∏–∫–ª–æ–º - –ø–æ–∏—Å–∫ –∫–Ω–∏–≥–∏ –ø–æ ISBN

                return book;
            }

    }
    
    // –ú–µ—Ç–æ–¥ —Å —Ü–∏–∫–ª–æ–º - –≤—Å–µ –¥–æ—Å—Ç—É–ø–Ω—ã–µ –∫–Ω–∏–≥–∏

        for (Book book : books) {
            if (book.isAvailable()) {  // –ø—Ä–∏–º–∏—Ç–∏–≤ boolean
                available.add(book);
            }
        }
        return available;
    }

                result.add(book);
            }

                             book.getTitle(), reader.getName());
    }

        book.returnItem();
        reader.removeBook(book);
        
        return String.format("–ö–Ω–∏–≥–∞ '%s' –≤–æ–∑–≤—Ä–∞—â–µ–Ω–∞", book.getTitle());
    }
    
    // –ú–µ—Ç–æ–¥ —Å —Ü–∏–∫–ª–æ–º - —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
    public void printStatistics() {

        int totalBooks = bookRepository.getTotalBooks();
        int availableBooks = allBooks.size();
        
        System.out.println("=== –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –±–∏–±–ª–∏–æ—Ç–µ–∫–∏ ===");

        System.out.println("–î–æ—Å—Ç—É–ø–Ω–æ: " + availableBooks);
        System.out.println("–í—ã–¥–∞–Ω–æ: " + (totalBooks - availableBooks));
    }
}
